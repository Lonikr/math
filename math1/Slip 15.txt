 3. Generate all prime numbers between 1000 to 2000 using Python program.
    
    def is_prime(n):
    if n <= 1:
        return False
    if n <= 3:
        return True
    if n % 2 == 0 or n % 3 == 0:
        return False
    i = 5
    while i * i <= n:
        if n % i == 0 or n % (i + 2) == 0:
            return False
        i += 6
    return True
start = 1000
end = 2000
prime_numbers = [x for x in range(start, end + 1) if is_prime(x)]
